\section{Collapse of the Full Yocto-Section Package}

\begin{proposition}[The minimizer-image body carries the body-level content]
\label{prop:body}
For every benchmark-faithful yocto-section package, \(\YoctoImgBody{\sigma}\) is compact and the restricted projection
\[
\YoctoImgProj{\sigma}:\YoctoImgBody{\sigma}\to\ZeptoSectionBody{\sigma}
\]
is a homeomorphism with inverse \(\YoctoSectionZeptoMin{\sigma}\).
\end{proposition}

\begin{proof}
The body \(\ZeptoSectionBody{\sigma}\) is compact by Definition \ref{def:zeptosec}, and \(\YoctoSectionZeptoMin{\sigma}\) is continuous by Definition \ref{def:yoctosec}(2). Therefore
\[
\YoctoImgBody{\sigma}
=
\YoctoSectionZeptoMin{\sigma}\bigl(\ZeptoSectionBody{\sigma}\bigr)
\]
is compact. The identity
\[
\YoctoSectionProj{\sigma}\circ\YoctoSectionZeptoMin{\sigma}
=
\mathrm{id}_{\ZeptoSectionBody{\sigma}}
\]
shows that \(\YoctoImgProj{\sigma}\) is surjective and that \(\YoctoSectionZeptoMin{\sigma}\) is a right inverse for it. If \(y_{1},y_{2}\in\YoctoImgBody{\sigma}\) satisfy
\[
\YoctoImgProj{\sigma}(y_{1})=\YoctoImgProj{\sigma}(y_{2})=q,
\]
then each \(y_{i}\) lies in the fiber \(\YoctoSectionFiber{\sigma}{q}\) and is the unique minimizer there, so \(y_{1}=y_{2}=\YoctoSectionZeptoMin{\sigma}(q)\). Thus \(\YoctoImgProj{\sigma}\) is bijective with inverse \(\YoctoSectionZeptoMin{\sigma}\). Since \(\YoctoImgBody{\sigma}\) is compact and \(\ZeptoSectionBody{\sigma}\) is Hausdorff, the continuous bijection \(\YoctoImgProj{\sigma}\) is a homeomorphism.
\end{proof}

\begin{proposition}[The recorded zepto-section functional is carried by the image core]
\label{prop:functional}
For every benchmark-faithful yocto-section package, the recorded zepto-section functional is recovered entirely from the restricted image-core functional:
\[

=
\YoctoImgFunctional{\sigma}\bigl(\YoctoSectionZeptoMin{\sigma}(q)\bigr)
\qquad
\text{for every }q\in\ZeptoSectionBody{\sigma}.
\]
\end{proposition}

\begin{proof}
By Definition \ref{def:imagecore}(2),
\[
\YoctoImgFunctional{\sigma}\bigl(\YoctoSectionZeptoMin{\sigma}(q)\bigr)
=
\YoctoSectionFunctional{\sigma}\bigl(\YoctoSectionZeptoMin{\sigma}(q)\bigr).
\]
Definition \ref{def:yoctosec}(2) then gives
\[
\YoctoSectionFunctional{\sigma}\bigl(\YoctoSectionZeptoMin{\sigma}(q)\bigr)
=
