\section{Canonical Recovery of the Current Zepto-Section Package}

\begin{proposition}[The canonical yocto minimizer section is well defined]
\label{prop:section}
For every benchmark-faithful yocto-section dynamics package, the map \(\YoctoSectionZeptoMin{\sigma}\) of Definition \ref{def:yoctosec} is a smooth section of \(\YoctoSectionProj{\sigma}\), so that
\[
\YoctoSectionProj{\sigma}\circ\YoctoSectionZeptoMin{\sigma}
=
\mathrm{id}_{\ZeptoSectionBody{\sigma}}.
\]
\end{proposition}

\begin{proof}
Definition \ref{def:yoctosec}(1) makes each fiber \(\YoctoSectionFiber{\sigma}{q}\) nonempty, compact, and convex. Definition \ref{def:yoctosec}(2) states that \(\YoctoSectionFunctional{\sigma}\) has a unique minimizer on each such fiber and that the minimizer depends smoothly on \(q\). Definition \ref{def:yoctosec}(2) therefore gives a well-defined smooth map
\[
\YoctoSectionZeptoMin{\sigma}:\ZeptoSectionBody{\sigma}\to\YoctoSectionBody{\sigma}.
\]
Because every minimizer lies in the fiber over the point at which it is selected, we have
\[
\YoctoSectionProj{\sigma}\bigl(\YoctoSectionZeptoMin{\sigma}(q)\bigr)=q
\]
for every \(q\in\ZeptoSectionBody{\sigma}\), which is the displayed identity.
\end{proof}

\begin{proposition}[The exact zepto-section shell is the projected exact yocto-section shell]
\label{prop:shell}
For every benchmark-faithful yocto-section dynamics package,
\[
\YoctoSectionProj{\sigma}\bigl(\YoctoSectionShell{\sigma}\bigr)
=
\ZeptoSectionShell{\sigma},
\]
and the shell restriction
\[
\YoctoSectionProj{\sigma}\big|_{\YoctoSectionShell{\sigma}}
:
\YoctoSectionShell{\sigma}\xrightarrow{\cong}\ZeptoSectionShell{\sigma}
\]
is a diffeomorphism with inverse
\[
\YoctoSectionZeptoMin{\sigma}\big|_{\ZeptoSectionShell{\sigma}}
:
\ZeptoSectionShell{\sigma}\xrightarrow{\cong}\YoctoSectionShell{\sigma}.
\]
\end{proposition}

\begin{proof}
Definition \ref{def:yoctosec}(3) states directly that
\[
\YoctoSectionProj{\sigma}\big|_{\YoctoSectionShell{\sigma}}
:
\YoctoSectionShell{\sigma}\xrightarrow{\cong}\ZeptoSectionShell{\sigma}
\]
is a diffeomorphism, so the image identity follows immediately. To identify the inverse, let \(q\in\ZeptoSectionShell{\sigma}\). By Proposition \ref{prop:section}, \(\YoctoSectionZeptoMin{\sigma}(q)\) lies in the fiber over \(q\). Since \(q\in\ZeptoSectionShell{\sigma}\), Definition \ref{def:yoctosec}(3) shows that this point belongs to \(\YoctoSectionShell{\sigma}\). Hence the shell restriction of \(\YoctoSectionZeptoMin{\sigma}\) is indeed the inverse diffeomorphism.
\end{proof}

\[

\YoctoSectionFunctional{\sigma}(y)
\qquad
\text{for every }

\begin{proof}
For each \(q\in\ZeptoSectionBody{\sigma}\), the point \(s(q)\) belongs to \(\YoctoSectionFiber{\sigma}{q}\) and minimizes \(\YoctoSectionFunctional{\sigma}\) there. By Definition \ref{def:yoctosec}(2), that minimizer is unique. Therefore
