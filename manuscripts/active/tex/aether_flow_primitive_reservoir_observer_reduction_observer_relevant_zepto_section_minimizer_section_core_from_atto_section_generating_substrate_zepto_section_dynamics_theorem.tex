\section{Canonical Recovery of the Current Section Core}

\begin{proposition}[The canonical minimizer section is well defined]
\label{prop:section}
For every benchmark-faithful zepto-section dynamics package, the map \(\ZeptoMinSection{\sigma}\) of Definition \ref{def:sectioncore} is a smooth section of \(\ZeptoSectionProj{\sigma}\), so that
\[
\ZeptoSectionProj{\sigma}\circ\ZeptoMinSection{\sigma}
=
\mathrm{id}_{\AttoSectionBody{\sigma}}.
\]
\end{proposition}

\begin{proof}
Definition \ref{def:zeptosec}(1) makes each fiber \(\ZeptoSectionFiber{\sigma}{v}\) nonempty, compact, and convex. Definition \ref{def:zeptosec}(2) states that \(\ZeptoSectionFunctional{\sigma}\) has a unique minimizer on each such fiber and that the minimizer depends smoothly on \(v\). Definition \ref{def:sectioncore}(1) therefore gives a well-defined smooth map
\[
\ZeptoMinSection{\sigma}:\AttoSectionBody{\sigma}\to\ZeptoSectionBody{\sigma}.
\]
Because every minimizer lies in the fiber over the point at which it is selected, we have
\[
\ZeptoSectionProj{\sigma}\bigl(\ZeptoMinSection{\sigma}(v)\bigr)=v
\]
for every \(v\in\AttoSectionBody{\sigma}\), which is the displayed identity.
\end{proof}

\begin{proposition}[The exact shell is the shell restriction of the canonical section]
\label{prop:shell}
For every benchmark-faithful zepto-section dynamics package,
\[
\ZeptoSectionShell{\sigma}
=
\ZeptoMinSection{\sigma}\bigl(\AttoSectionShell{\sigma}\bigr),
\]
and the shell restriction
\[
\ZeptoMinSection{\sigma}\big|_{\AttoSectionShell{\sigma}}
:
\AttoSectionShell{\sigma}\xrightarrow{\cong}\ZeptoSectionShell{\sigma}
\]
is a diffeomorphism with inverse
\[
\ZeptoSectionProj{\sigma}\big|_{\ZeptoSectionShell{\sigma}}
:
\ZeptoSectionShell{\sigma}\xrightarrow{\cong}\AttoSectionShell{\sigma}.
\]
\end{proposition}

\begin{proof}
Let \(v\in\AttoSectionShell{\sigma}\). By Definition \ref{def:sectioncore}(1), \(\ZeptoMinSection{\sigma}(v)\) is the unique minimizer of \(\ZeptoSectionFunctional{\sigma}\) on the fiber \(\ZeptoSectionFiber{\sigma}{v}\). Since \(v\in\AttoSectionShell{\sigma}\), Definition \ref{def:zeptosec}(3) implies
\[
\ZeptoMinSection{\sigma}(v)\in\ZeptoSectionShell{\sigma}.
\]
Hence
\[
\ZeptoMinSection{\sigma}\bigl(\AttoSectionShell{\sigma}\bigr)\subset\ZeptoSectionShell{\sigma}.
\]

Conversely, if \(q\in\ZeptoSectionShell{\sigma}\), then \(v\equiv\ZeptoSectionProj{\sigma}(q)\) lies in \(\AttoSectionShell{\sigma}\), and Definition \ref{def:zeptosec}(3) says that \(q\) is a minimizer of \(\ZeptoSectionFunctional{\sigma}\) on \(\ZeptoSectionFiber{\sigma}{v}\). By uniqueness of that minimizer, \(q=\ZeptoMinSection{\sigma}(v)\). This proves the displayed equality.

The map \(\ZeptoSectionProj{\sigma}\big|_{\ZeptoSectionShell{\sigma}}\) is a diffeomorphism by Definition \ref{def:zeptosec}(3). The displayed equality shows that its inverse is exactly \(\ZeptoMinSection{\sigma}\big|_{\AttoSectionShell{\sigma}}\).
\end{proof}
