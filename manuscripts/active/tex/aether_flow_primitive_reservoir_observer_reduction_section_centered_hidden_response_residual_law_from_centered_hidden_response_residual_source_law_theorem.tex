\begin{definition}[Minimal residual-source lift]
\label{def:lift}
Fix \(\sigma\in\{\Car,\Cur,\Hom\}\) and a benchmark-faithful centered hidden-response residual-source law. For \(y\in\ProtoSectionBody{\sigma}\), define the residual-source quotient metric
\begin{equation}
\ResidualSourceQuotientMetric{\sigma}(y)
\equiv
\ResidualSourceProjection{\sigma}(y)
\ResidualSourceOperator{\sigma}(y)^{-1}
\ResidualSourceProjection{\sigma}(y)^*.
\label{eq:quotientmetric}
\end{equation}
Also define the minimal residual-source lift by
\begin{equation}
\begin{aligned}
\ResidualSourceLift{\sigma}(y)
\equiv
\ResidualSourceOperator{\sigma}(y)^{-1}
\ResidualSourceProjection{\sigma}(y)^*
\ResidualSourceQuotientMetric{\sigma}(y)^{-1}
:
\ResidualState{\sigma}\to\ResidualSourceCarrier{\sigma}.
\end{aligned}
\label{eq:minlift}
\end{equation}
Finally define the residual-source energy
\[
\ResidualSourceEnergy{\sigma}(y,w)
\equiv
\frac{1}{2}
\ev{
w,\ResidualSourceOperator{\sigma}(y)w
},
\qquad
w\in\ResidualSourceCarrier{\sigma}.
\]
\end{definition}

\begin{proposition}[The residual-source quotient metric is positive definite]
\label{prop:quotientmetric}
For every \(y\in\ProtoSectionBody{\sigma}\), the operator \(\ResidualSourceQuotientMetric{\sigma}(y)\) of \eqref{eq:quotientmetric} is positive self-adjoint and positive definite on \(\ResidualState{\sigma}\).
\end{proposition}

\begin{proof}
Self-adjointness is immediate from the definition. For any \(u\in\ResidualState{\sigma}\),
\[
\ev{
u,\ResidualSourceQuotientMetric{\sigma}(y)u
}
=
\ev{
\ResidualSourceOperator{\sigma}(y)^{-1/2}
\ResidualSourceProjection{\sigma}(y)^*u,
\ResidualSourceOperator{\sigma}(y)^{-1/2}
\ResidualSourceProjection{\sigma}(y)^*u
}.
\]
The right-hand side is nonnegative and vanishes only if
\[
\ResidualSourceProjection{\sigma}(y)^*u=0.
\]
Because \(\ResidualSourceProjection{\sigma}(y)\) is surjective, its adjoint is injective, so \(\ResidualSourceProjection{\sigma}(y)^*u=0\) implies \(u=0\). Hence \(\ResidualSourceQuotientMetric{\sigma}(y)\) is positive definite.
\end{proof}

\begin{proposition}[Minimal-source-energy characterization]
\label{prop:minlift}
For every \(y\in\ProtoSectionBody{\sigma}\), the map \(\ResidualSourceLift{\sigma}(y)\) of \eqref{eq:minlift} is the unique linear right inverse of \(\ResidualSourceProjection{\sigma}(y)\) whose value at \(u\in\ResidualState{\sigma}\) minimizes the residual-source energy on the affine fiber
\[
\left\{
w\in\ResidualSourceCarrier{\sigma}:
\ResidualSourceProjection{\sigma}(y)w=u
\right\}.
\]
More precisely,
\begin{equation}
\ResidualSourceProjection{\sigma}(y)\ResidualSourceLift{\sigma}(y)
=
\mathrm{id}_{\ResidualState{\sigma}},
\label{eq:rightinverse}
\end{equation}
and for every \(u\in\ResidualState{\sigma}\),
\begin{equation}
\ResidualSourceEnergy{\sigma}\bigl(y,\ResidualSourceLift{\sigma}(y)u\bigr)
=
\min_{\ResidualSourceProjection{\sigma}(y)w=u}
\ResidualSourceEnergy{\sigma}(y,w).
\label{eq:minenergy}
\end{equation}
\end{proposition}

\begin{proof}
Equation \eqref{eq:rightinverse} follows from Definitions \ref{def:lift} and Proposition \ref{prop:quotientmetric}:
\[
\begin{aligned}
\ResidualSourceProjection{\sigma}(y)\ResidualSourceLift{\sigma}(y)
&=
\ResidualSourceProjection{\sigma}(y)
\ResidualSourceOperator{\sigma}(y)^{-1}
\ResidualSourceProjection{\sigma}(y)^*
\ResidualSourceQuotientMetric{\sigma}(y)^{-1}\\
&=
\ResidualSourceQuotientMetric{\sigma}(y)
\ResidualSourceQuotientMetric{\sigma}(y)^{-1}
=
\mathrm{id}.
\end{aligned}
\]
Fix \(u\in\ResidualState{\sigma}\) and write
\[
w=\ResidualSourceLift{\sigma}(y)u+\eta,
\qquad
\eta\in\ker \ResidualSourceProjection{\sigma}(y).
\]
Because \(\ResidualSourceLift{\sigma}(y)\) is built from the Euler-Lagrange formula for the constrained quadratic problem, we have
\[
\begin{aligned}
\ev{
\ResidualSourceOperator{\sigma}(y)\ResidualSourceLift{\sigma}(y)u,
\eta
}
&=
\ev{
\ResidualSourceProjection{\sigma}(y)^*
\ResidualSourceQuotientMetric{\sigma}(y)^{-1}u,
\eta
}\\
&=
\ev{
\ResidualSourceQuotientMetric{\sigma}(y)^{-1}u,
\ResidualSourceProjection{\sigma}(y)\eta
}
=
0.
\end{aligned}
\]
Therefore
\[
\ResidualSourceEnergy{\sigma}(y,w)
=
\ResidualSourceEnergy{\sigma}\bigl(y,\ResidualSourceLift{\sigma}(y)u\bigr)
+
\frac{1}{2}
\ev{
\eta,\ResidualSourceOperator{\sigma}(y)\eta
}.
\]
The second term is nonnegative and vanishes only when \(\eta=0\) because \(\ResidualSourceOperator{\sigma}(y)\) is positive definite. This proves \eqref{eq:minenergy} and uniqueness of the minimizer.
\end{proof}

\begin{proposition}[The source displacement readout descends to the residual state space]
\label{prop:descent}

\[

\end{equation}

\begin{equation}

=
\ResidualSourceReadout{\sigma}(y)\ResidualSourceLift{\sigma}(y).
\label{eq:residualreadout}
\end{equation}
\end{proposition}

\begin{proof}
The inclusion
\[

\subseteq
\ker \ResidualSourceReadout{\sigma}(y)
\]
means that \(\ResidualSourceReadout{\sigma}(y)\) is constant on the fibers of \(\ResidualSourceProjection{\sigma}(y)\). Surjectivity of \(\ResidualReadout{\sigma}(y)\) follows from surjectivity of \(\ResidualSourceReadout{\sigma}(y)\) and \(\ResidualSourceProjection{\sigma}(y)\). Equation \eqref{eq:residualreadout} is immediate from \eqref{eq:descent} and \eqref{eq:rightinverse}. Smoothness follows because both source maps and \(\ResidualSourceLift{\sigma}(y)\) vary smoothly in \(y\).
\end{proof}

\end{proof}

\begin{proof}
