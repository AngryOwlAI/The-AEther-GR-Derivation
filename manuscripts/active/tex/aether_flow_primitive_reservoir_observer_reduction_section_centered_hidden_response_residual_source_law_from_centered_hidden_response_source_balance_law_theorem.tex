\section{Induced Centered Hidden-Response Residual-Source Law}

\begin{definition}[Balanced kernel and response-orthogonal balance projector]
\label{def:projector}
Fix \(\sigma\in\{\Car,\Cur,\Hom\}\) and a benchmark-faithful centered hidden-response source-balance law. Define the balanced source state space
\[
\SourceBalancedState{\sigma}
\equiv
\ker \SourceBalanceOperator{\sigma}.
\]
For \(y\in\ProtoSectionBody{\sigma}\), define the source-balance quotient metric
\begin{equation}
\SourceBalanceQuotientMetric{\sigma}(y)
\equiv
\SourceBalanceOperator{\sigma}
\SourceBalanceResponse{\sigma}(y)^{-1}
{\SourceBalanceOperator{\sigma}}^{*}.
\label{eq:balancequotient}
\end{equation}
Also define the response-orthogonal balance corrector
\begin{equation}
\SourceBalanceCorrector{\sigma}(y)
\equiv
\SourceBalanceResponse{\sigma}(y)^{-1}
{\SourceBalanceOperator{\sigma}}^{*}
\SourceBalanceQuotientMetric{\sigma}(y)^{-1}
:
\SourceBalanceTarget{\sigma}\to\BalancedSourceCarrier{\sigma}
\label{eq:corrector}
\end{equation}
and the balanced projection
\begin{equation}
\BalancedProjection{\sigma}(y)
\equiv
I_{\BalancedSourceCarrier{\sigma}}
-\SourceBalanceCorrector{\sigma}(y)\SourceBalanceOperator{\sigma}.
\label{eq:projector}
\end{equation}
Finally, for \(w\in\BalancedSourceCarrier{\sigma}\) and \(u\in\SourceBalancedState{\sigma}\), define the source-balance correction energy
\[
\SourceBalanceCorrectionEnergy{\sigma}(y;w,u)
\equiv
\frac{1}{2}
\ev{
w-u,
\SourceBalanceResponse{\sigma}(y)(w-u)
}.
\]
\end{definition}

\begin{proposition}[The source-balance quotient metric is positive definite]
\label{prop:balancequotient}
For every \(y\in\ProtoSectionBody{\sigma}\), the operator \(\SourceBalanceQuotientMetric{\sigma}(y)\) of \eqref{eq:balancequotient} is positive self-adjoint and positive definite on \(\SourceBalanceTarget{\sigma}\).
\end{proposition}

\begin{proof}
Self-adjointness is immediate from the definition. For any \(c\in\SourceBalanceTarget{\sigma}\),
\[
\ev{
c,\SourceBalanceQuotientMetric{\sigma}(y)c
}
=
\ev{
\SourceBalanceResponse{\sigma}(y)^{-1/2}
{\SourceBalanceOperator{\sigma}}^{*}c,
\SourceBalanceResponse{\sigma}(y)^{-1/2}
{\SourceBalanceOperator{\sigma}}^{*}c
}.
\]
The right-hand side is nonnegative and vanishes only if
\[
{\SourceBalanceOperator{\sigma}}^{*}c=0.
\]
Because \(\SourceBalanceOperator{\sigma}\) is surjective, its adjoint is injective, so \({\SourceBalanceOperator{\sigma}}^{*}c=0\) implies \(c=0\). Hence \(\SourceBalanceQuotientMetric{\sigma}(y)\) is positive definite.
\end{proof}

\begin{proposition}[The balanced projection is the canonical projector onto the balanced kernel]
\label{prop:projector}
For every \(y\in\ProtoSectionBody{\sigma}\), the map \(\BalancedProjection{\sigma}(y)\) of \eqref{eq:projector} is a surjective idempotent linear map onto \(\SourceBalancedState{\sigma}\). More precisely,
\begin{enumerate}
    \item
    \[
    \SourceBalanceOperator{\sigma}\BalancedProjection{\sigma}(y)=0;
    \]
    \item
    \[
    \BalancedProjection{\sigma}(y)u=u
    \qquad
    \text{for every }
    u\in\SourceBalancedState{\sigma};
    \]
    \item
    \[
    \BalancedProjection{\sigma}(y)^{2}
    =
    \BalancedProjection{\sigma}(y);
    \]
    \item
    \[
    \ker \BalancedProjection{\sigma}(y)
    =
    \operatorname{im}\SourceBalanceCorrector{\sigma}(y).
    \]
\end{enumerate}
\end{proposition}

\begin{proof}
Because
\[
\SourceBalanceOperator{\sigma}\SourceBalanceCorrector{\sigma}(y)
=
\SourceBalanceOperator{\sigma}
\SourceBalanceResponse{\sigma}(y)^{-1}
{\SourceBalanceOperator{\sigma}}^{*}
\SourceBalanceQuotientMetric{\sigma}(y)^{-1}
=
I_{\SourceBalanceTarget{\sigma}},
\]
we get
\[
\SourceBalanceOperator{\sigma}\BalancedProjection{\sigma}(y)
=
\SourceBalanceOperator{\sigma}
-\SourceBalanceOperator{\sigma}\SourceBalanceCorrector{\sigma}(y)\SourceBalanceOperator{\sigma}
=
0.
\]
Hence \(\operatorname{im}\BalancedProjection{\sigma}(y)\subseteq\SourceBalancedState{\sigma}\).

If \(u\in\SourceBalancedState{\sigma}\), then \(\SourceBalanceOperator{\sigma}u=0\), so
\[
\BalancedProjection{\sigma}(y)u
=
u-\SourceBalanceCorrector{\sigma}(y)\SourceBalanceOperator{\sigma}u
=
u.
\]
Therefore \(\BalancedProjection{\sigma}(y)\) is surjective onto \(\SourceBalancedState{\sigma}\), and idempotence follows:
\[
\BalancedProjection{\sigma}(y)^{2}w
=
\BalancedProjection{\sigma}(y)\bigl(\BalancedProjection{\sigma}(y)w\bigr)
=
\BalancedProjection{\sigma}(y)w.
\]

If \(w\in\ker \BalancedProjection{\sigma}(y)\), then
\[
w
=
\SourceBalanceCorrector{\sigma}(y)\SourceBalanceOperator{\sigma}w
\in
\operatorname{im}\SourceBalanceCorrector{\sigma}(y).
\]
Conversely, if \(w=\SourceBalanceCorrector{\sigma}(y)c\), then
\[
\BalancedProjection{\sigma}(y)w
=
\SourceBalanceCorrector{\sigma}(y)c
-\SourceBalanceCorrector{\sigma}(y)\SourceBalanceOperator{\sigma}\SourceBalanceCorrector{\sigma}(y)c
=
0.
\]
Thus \(\ker \BalancedProjection{\sigma}(y)=\operatorname{im}\SourceBalanceCorrector{\sigma}(y)\).
\end{proof}

\begin{proposition}[Minimal source-balance correction characterization]
\label{prop:minimal}
For every \(y\in\ProtoSectionBody{\sigma}\) and every \(w\in\BalancedSourceCarrier{\sigma}\), the element \(\BalancedProjection{\sigma}(y)w\) is the unique vector in \(\SourceBalancedState{\sigma}\) minimizing the source-balance correction energy:
\begin{equation}
\SourceBalanceCorrectionEnergy{\sigma}\bigl(y;w,\BalancedProjection{\sigma}(y)w\bigr)
=
\min_{u\in\SourceBalancedState{\sigma}}
\SourceBalanceCorrectionEnergy{\sigma}(y;w,u).
\label{eq:minimalcorrection}
\end{equation}
\end{proposition}

\begin{proof}
Fix \(y\in\ProtoSectionBody{\sigma}\) and \(w\in\BalancedSourceCarrier{\sigma}\). By Proposition \ref{prop:projector}, the vector \(\BalancedProjection{\sigma}(y)w\) lies in \(\SourceBalancedState{\sigma}\). Any other \(u\in\SourceBalancedState{\sigma}\) can be written uniquely as
\[
u=\BalancedProjection{\sigma}(y)w+\eta,
\qquad
\eta\in\SourceBalancedState{\sigma}.
\]
Hence
\[
w-u
=
\SourceBalanceCorrector{\sigma}(y)\SourceBalanceOperator{\sigma}w-\eta.
\]
Using
\[
\SourceBalanceResponse{\sigma}(y)\SourceBalanceCorrector{\sigma}(y)
=
{\SourceBalanceOperator{\sigma}}^{*}
\SourceBalanceQuotientMetric{\sigma}(y)^{-1},
\]
we obtain
\[
\begin{aligned}
\ev{
\SourceBalanceCorrector{\sigma}(y)\SourceBalanceOperator{\sigma}w,
\SourceBalanceResponse{\sigma}(y)\eta
}
&=
\ev{
\SourceBalanceQuotientMetric{\sigma}(y)^{-1}\SourceBalanceOperator{\sigma}w,
\SourceBalanceOperator{\sigma}\eta
}\\
&=
0,
\end{aligned}
\]
because \(\eta\in\ker \SourceBalanceOperator{\sigma}\). Therefore
\[
\begin{aligned}
\SourceBalanceCorrectionEnergy{\sigma}(y;w,u)
&=
\SourceBalanceCorrectionEnergy{\sigma}\bigl(y;w,\BalancedProjection{\sigma}(y)w\bigr)\\
&\qquad+
\frac{1}{2}
\ev{
\eta,\SourceBalanceResponse{\sigma}(y)\eta
}.
\end{aligned}
\]
The second term is nonnegative and vanishes only when \(\eta=0\) because \(\SourceBalanceResponse{\sigma}(y)\) is positive definite. This proves \eqref{eq:minimalcorrection} and uniqueness of the minimizer.
\end{proof}

\begin{proposition}[The source displacement readout is exact on balanced-source fibers]
\label{prop:readout}
Under Definition \ref{def:balance}, for every \(y\in\ProtoSectionBody{\sigma}\),
\begin{equation}
\SourceBalanceReadout{\sigma}(y)\BalancedProjection{\sigma}(y)

\label{eq:readoutexact}
\end{equation}
Equivalently,
\[

\subseteq
\ker \SourceBalanceReadout{\sigma}(y).
\]
\end{proposition}

\begin{proof}
By Definition \ref{def:balance}(7),
