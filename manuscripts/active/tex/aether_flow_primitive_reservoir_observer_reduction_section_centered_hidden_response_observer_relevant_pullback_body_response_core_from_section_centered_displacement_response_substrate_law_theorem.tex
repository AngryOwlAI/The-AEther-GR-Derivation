The active derivational program of \TheoryName{} now has a sharper burden than before.     \item A finite-dimensional Euclidean substrate carrier

    \[
    \SubstrateBody{\sigma}
    \subset
    \ProtoSectionBody{\sigma}\times\SubstrateCarrier{\sigma}

    \[
    \SubstrateBody{\sigma}(y)
    \equiv
    \left\{
    u\in\SubstrateCarrier{\sigma}:(y,u)\in\SubstrateBody{\sigma}
    \right\}
    \]
    is nonempty, compact, convex, contains \(0\), and has nonempty interior for every \(y\in\ProtoSectionBody{\sigma}\).
    \item A smooth family of surjective linear displacement projections
    \[
    \SubstrateProjection{\sigma}(y):
    \SubstrateCarrier{\sigma}\to\PrePreProtoSectionVertical{\sigma},
    \qquad
    y\in\ProtoSectionBody{\sigma}.
    \]
    \item A smooth family of positive self-adjoint substrate response operators
    \[
    \SubstrateOperator{\sigma}(y):
    \SubstrateCarrier{\sigma}\to\SubstrateCarrier{\sigma},
    \qquad
    y\in\ProtoSectionBody{\sigma},
    \]

    \begin{equation}
    \ev{u,\SubstrateOperator{\sigma}(y)u}
    \ge
    \mu_{\sigma}\,\|u\|^{2}
    \qquad
    \text{for all }
    y\in\ProtoSectionBody{\sigma},
    \ \ u\in\SubstrateCarrier{\sigma}
    \label{eq:coercive}
    \end{equation}
    for some constant \(\mu_{\sigma}>0\).
    \item A smooth family of linear minimal lifts
    \[
    \SubstrateLift{\sigma}(y):
    \PrePreProtoSectionVertical{\sigma}\to\SubstrateCarrier{\sigma}
    \]
    satisfying, for every \(y\in\ProtoSectionBody{\sigma}\),

        \SubstrateProjection{\sigma}(y)\circ\SubstrateLift{\sigma}(y)

        \]
        \item every \(u\in\SubstrateCarrier{\sigma}\) admits the unique decomposition
        \[
        u
        =
        \SubstrateLift{\sigma}(y)\bigl(\SubstrateProjection{\sigma}(y)u\bigr)

        k,
        \qquad
        k\in\SubstrateKernel{\sigma}(y)\equiv\ker\SubstrateProjection{\sigma}(y);
        \]
        \item the lift is \(\SubstrateOperator{\sigma}(y)\)-orthogonal to the kernel, namely
        \[
        \ev{
        \SubstrateLift{\sigma}(y)v,

\section{Forced Body-Restricted Pullback Response}

\begin{definition}[Substrate response energy and law-forced body response]
\label{def:forced}
Fix \(\sigma\in\{\Car,\Cur,\Hom\}\) and a benchmark-faithful section-centered displacement-response substrate law. Define the quadratic substrate response energy by
\[
\SubstrateEnergy{\sigma}(y,u)
\equiv
\frac{1}{2}
\ev{u,\SubstrateOperator{\sigma}(y)u}.
\]
For \(v\in\SubstrateProjection{\sigma}(y)(\SubstrateCarrier{\sigma})\), define the \emph{law-forced body response} by
\begin{equation}
\ForcedBodyResponse{\sigma}(y,v)
\equiv
\min\left\{
\SubstrateEnergy{\sigma}(y,u):
u\in\SubstrateCarrier{\sigma},
\ \SubstrateProjection{\sigma}(y)u=v
\right\}.
\label{eq:forcedresponse}
\end{equation}
\end{definition}

\begin{proposition}[Unique minimal substrate lift]
\label{prop:minlift}
Fix \(y\in\ProtoSectionBody{\sigma}\) and \(v\in\PrePreProtoSectionVertical{\sigma}\). For every benchmark-faithful section-centered displacement-response substrate law, the constrained minimum in \eqref{eq:forcedresponse} exists, is unique, and is achieved by the lifted substrate state
\[
u_{\min}(y,v)
=
\SubstrateLift{\sigma}(y)v.
\]
Consequently,
\begin{equation}
\ForcedBodyResponse{\sigma}(y,v)
=
\frac{1}{2}
\ev{
\SubstrateLift{\sigma}(y)v,
\SubstrateOperator{\sigma}(y)\SubstrateLift{\sigma}(y)v
}.
\label{eq:forcedquadratic}
\end{equation}
\end{proposition}

\begin{proof}
Let \(u\in\SubstrateCarrier{\sigma}\) satisfy \(\SubstrateProjection{\sigma}(y)u=v\). By Definition \ref{def:law}(4), \(u\) admits the unique decomposition
\[
u
=
\SubstrateLift{\sigma}(y)v+k,
\qquad
k\in\SubstrateKernel{\sigma}(y).
\]
Using the \(\SubstrateOperator{\sigma}(y)\)-orthogonality of the lift to the kernel,
\[
\ev{u,\SubstrateOperator{\sigma}(y)u}
=
\ev{
\SubstrateLift{\sigma}(y)v,
\SubstrateOperator{\sigma}(y)\SubstrateLift{\sigma}(y)v
}
+
\ev{k,\SubstrateOperator{\sigma}(y)k}.
\]
By the coercive bound \eqref{eq:coercive}, the second term is nonnegative and vanishes only when \(k=0\). Hence the minimum among all lifts of \(v\) is attained uniquely at \(k=0\), namely at \(u=\SubstrateLift{\sigma}(y)v\). Formula \eqref{eq:forcedquadratic} follows immediately.
\end{proof}

\begin{proposition}[The law-forced response is a benchmark-faithful body response]
\label{prop:response}
Fix a benchmark-faithful current-body-realizing section-centered displacement-response substrate law for sector \(\sigma\). Then the pair
\[
\left(
\PrePreProtoSectionResidualBody{\sigma},
\ForcedBodyResponse{\sigma}
\right)
\]
is a benchmark-faithful observer-relevant pullback body-response core over the fixed proto-section benchmark base.
\end{proposition}

\begin{proof}
Because the law is current-body-realizing, the recorded displacement body is exactly the projection image \eqref{eq:realizing} of the compact convex substrate body. Hence each fiber \(\PrePreProtoSectionResidualBody{\sigma}(y)\) is the image of a nonempty compact convex set under a linear map, and so remains nonempty, compact, and convex; its nonempty interior and zero section are already part of the recorded current-body datum. By Proposition \ref{prop:minlift}, the forced response \(\ForcedBodyResponse{\sigma}(y,v)\) is a smooth fiberwise quadratic form in \(v\). If \(v\neq 0\), then \(\SubstrateLift{\sigma}(y)v\neq 0\) because \(\SubstrateProjection{\sigma}(y)\circ\SubstrateLift{\sigma}(y)=\mathrm{id}\), and the positivity of \(\SubstrateOperator{\sigma}(y)\) gives
\[
\ForcedBodyResponse{\sigma}(y,v)>0.
\]
Conversely \(\ForcedBodyResponse{\sigma}(y,0)=0\). Benchmark faithfulness is part of Definition \ref{def:law}(6). Therefore the induced pair satisfies Definition \ref{def:bodyresp}.
\end{proof}
