\section{Collapse of the Image Core}

\begin{proposition}[The canonical minimizer section reconstructs the image body and projection]
\label{prop:reconstructbody}
For every benchmark-faithful yocto-section package,
\[
\YoctoImgBody{\sigma}
=
\YoctoMinSection{\sigma}\bigl(\ZeptoSectionBody{\sigma}\bigr),
\]
and the restricted projection
\[
\YoctoImgProj{\sigma}:\YoctoImgBody{\sigma}\to\ZeptoSectionBody{\sigma}
\]
is a homeomorphism with inverse \(\YoctoMinSection{\sigma}\).
\end{proposition}

\begin{proof}
The displayed identity is Definition \ref{def:imagecore}(1) together with Definition \ref{def:sectioncore}(1). By Definition \ref{def:yoctosec}(2),
\[
\YoctoSectionProj{\sigma}\circ\YoctoMinSection{\sigma}
=
\mathrm{id}_{\ZeptoSectionBody{\sigma}}.
\]
Restricting \(\YoctoSectionProj{\sigma}\) to \(\YoctoImgBody{\sigma}\) yields
\[
\YoctoImgProj{\sigma}\circ\YoctoMinSection{\sigma}
=
\mathrm{id}_{\ZeptoSectionBody{\sigma}}.
\]
Conversely, every \(y\in\YoctoImgBody{\sigma}\) has the form
\[
y=\YoctoMinSection{\sigma}(q)
\qquad
\text{for some }
q\in\ZeptoSectionBody{\sigma}.
\]
Hence
\[
\YoctoMinSection{\sigma}\bigl(\YoctoImgProj{\sigma}(y)\bigr)
=
\YoctoMinSection{\sigma}\bigl(\YoctoSectionProj{\sigma}(y)\bigr)
=
\YoctoMinSection{\sigma}\bigl(\YoctoSectionProj{\sigma}(\YoctoMinSection{\sigma}(q))\bigr)
=
\YoctoMinSection{\sigma}(q)
=
y.
\]
Therefore \(\YoctoImgProj{\sigma}\) and \(\YoctoMinSection{\sigma}\) are mutually inverse continuous maps, so the restricted projection is a homeomorphism with inverse \(\YoctoMinSection{\sigma}\).
\end{proof}

\begin{proposition}[The canonical minimizer section reconstructs the image-core functional]
\label{prop:reconstructfunctional}
For every benchmark-faithful yocto-section package, the restricted image-core functional is recovered from the canonical minimizer section and the fixed zepto-section functional by
\[
\YoctoImgFunctional{\sigma}\bigl(\YoctoMinSection{\sigma}(q)\bigr)
=
\ZeptoSectionFunctional{\sigma}(q)
\qquad
\text{for every }q\in\ZeptoSectionBody{\sigma}.
\]
Equivalently,
\[
\YoctoImgFunctional{\sigma}(y)
=
\ZeptoSectionFunctional{\sigma}\bigl((\YoctoMinSection{\sigma})^{-1}(y)\bigr)
\qquad
\text{for every }y\in\YoctoImgBody{\sigma}.
\]
\end{proposition}

\begin{proof}
By Definition \ref{def:imagecore}(2),
\[
\YoctoImgFunctional{\sigma}\bigl(\YoctoMinSection{\sigma}(q)\bigr)
=
\YoctoSectionFunctional{\sigma}\bigl(\YoctoSectionZeptoMin{\sigma}(q)\bigr).
\]
Definition \ref{def:yoctosec}(2) then gives
\[
\YoctoSectionFunctional{\sigma}\bigl(\YoctoSectionZeptoMin{\sigma}(q)\bigr)
=
\ZeptoSectionFunctional{\sigma}(q)
\]
for every \(q\in\ZeptoSectionBody{\sigma}\). This proves the first displayed identity. The equivalent formula on \(\YoctoImgBody{\sigma}\) follows from Proposition \ref{prop:reconstructbody}, which shows that every \(y\in\YoctoImgBody{\sigma}\) has the unique form \(y=\YoctoMinSection{\sigma}(q)\).
\end{proof}

\begin{proposition}[The shell image is exactly the shell restriction of the canonical minimizer section]
\label{prop:reconstructshell}
For every benchmark-faithful yocto-section package,
\[
\YoctoImgShell{\sigma}
=

\]

\YoctoImgProj{\sigma}\big|_{\YoctoImgShell{\sigma}}

\end{proposition}

\begin{proof}
The displayed equality is Definition \ref{def:imagecore}(3) together with Definition \ref{def:sectioncore}(1). By Definition \ref{def:yoctosec}(3),
\[
\YoctoSectionProj{\sigma}\big|_{\YoctoSectionShell{\sigma}}
:
\YoctoSectionShell{\sigma}\xrightarrow{\cong}\ZeptoSectionShell{\sigma}
\]
is a diffeomorphism, and
\[
\YoctoMinSection{\sigma}\bigl(\ZeptoSectionShell{\sigma}\bigr)
\subset
\YoctoSectionShell{\sigma}.
\]
Restricting the shell diffeomorphism to \(\YoctoImgShell{\sigma}\) gives
\[
\YoctoImgProj{\sigma}\big|_{\YoctoImgShell{\sigma}}
\circ
\YoctoMinSection{\sigma}\big|_{\ZeptoSectionShell{\sigma}}

\mathrm{id}_{\ZeptoSectionShell{\sigma}}.
\]
Conversely, if \(z\in\YoctoImgShell{\sigma}\), then \(z=\YoctoMinSection{\sigma}(q)\) for some \(q\in\ZeptoSectionShell{\sigma}\), so
\[
\YoctoMinSection{\sigma}\bigl(\YoctoImgProj{\sigma}(z)\bigr)
=
z.
\]
Thus the shell restriction of the canonical minimizer section and the restricted shell projection are inverse diffeomorphisms.
\end{proof}

\begin{proposition}[The full image core is reconstructed from the section core]
\label{prop:reconstructimage}
