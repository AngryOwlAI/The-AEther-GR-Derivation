\section{Canonical Visible-Response Representative}

\begin{proposition}[Positive Schur-response operator]
\label{prop:schur}
Fix a benchmark-faithful current-body-realizing projection-operator core in the sense of Definition \ref{def:core}. For each \(y\in\ProtoSectionBody{\sigma}\), define
\[
\SchurResponse{\sigma}(y)
\equiv
\SubstrateProjection{\sigma}(y)
\SubstrateOperator{\sigma}(y)^{-1}
\SubstrateProjection{\sigma}(y)^*:
\PrePreProtoSectionVertical{\sigma}\to\PrePreProtoSectionVertical{\sigma},
\]
where \((\cdot)^*\) denotes Euclidean adjoint with respect to the fixed inner products on \(\SubstrateCarrier{\sigma}\) and \(\PrePreProtoSectionVertical{\sigma}\). Then \(\SchurResponse{\sigma}(y)\) is positive self-adjoint and invertible for every \(y\in\ProtoSectionBody{\sigma}\).
\end{proposition}

\begin{proof}
Self-adjointness is immediate from the self-adjointness of \(\SubstrateOperator{\sigma}(y)^{-1}\). For any \(0\neq w\in\PrePreProtoSectionVertical{\sigma}\),
\[
\ev{w,\SchurResponse{\sigma}(y)w}
=
\ev{
\SubstrateOperator{\sigma}(y)^{-1/2}\SubstrateProjection{\sigma}(y)^*w,
\SubstrateOperator{\sigma}(y)^{-1/2}\SubstrateProjection{\sigma}(y)^*w
}
>0
\]
because \(\SubstrateProjection{\sigma}(y)\) is surjective and hence \(\SubstrateProjection{\sigma}(y)^*\) is injective. Thus \(\SchurResponse{\sigma}(y)\) is positive definite and invertible on the finite-dimensional space \(\PrePreProtoSectionVertical{\sigma}\).
\end{proof}

\begin{proposition}[Canonical lift and energy decomposition]
\label{prop:decomp}
Fix a benchmark-faithful current-body-realizing projection-operator core. Then, for every \(y\in\ProtoSectionBody{\sigma}\),
\begin{enumerate}
    \item
    \[
    \SubstrateProjection{\sigma}(y)\circ\CanonicalLift{\sigma}(y)
    =
    \mathrm{id}_{\PrePreProtoSectionVertical{\sigma}};
    \]
    \item
    \[
    \ev{
    \CanonicalLift{\sigma}(y)v,
    \SubstrateOperator{\sigma}(y)k
    }
    =
    0
    \]
    for every \(v\in\PrePreProtoSectionVertical{\sigma}\) and every \(k\in\SubstrateKernel{\sigma}(y)\equiv\ker\SubstrateProjection{\sigma}(y)\);
    \item every \(u\in\SubstrateCarrier{\sigma}\) admits the unique decomposition
    \[
    u
    =
    \CanonicalLift{\sigma}(y)\bigl(\SubstrateProjection{\sigma}(y)u\bigr)
    +
    k,
    \qquad
    k\in\SubstrateKernel{\sigma}(y);
    \]
    \item for \(u\in\SubstrateCarrier{\sigma}\), with
    \[
    v\equiv\SubstrateProjection{\sigma}(y)u,
    \qquad
    k\equiv u-\CanonicalLift{\sigma}(y)v,
    \]
    the energy splits as
    \begin{equation}
    \ev{u,\SubstrateOperator{\sigma}(y)u}
    =
    \ev{v,\VisibleResponseOperator{\sigma}(y)v}
    +
    \ev{k,\SubstrateOperator{\sigma}(y)k}.
    \label{eq:split}
    \end{equation}
\end{enumerate}
\end{proposition}

\begin{proof}
Item (1) follows directly from Definitions \ref{def:visible} and \ref{prop:schur}:
\[
\SubstrateProjection{\sigma}(y)\CanonicalLift{\sigma}(y)
=
\SchurResponse{\sigma}(y)\VisibleResponseOperator{\sigma}(y)
=
\mathrm{id}.
\]
For item (2), let \(k\in\SubstrateKernel{\sigma}(y)\). Then
\begin{align*}
\ev{
\CanonicalLift{\sigma}(y)v,
\SubstrateOperator{\sigma}(y)k
}
&=
\ev{
\SubstrateProjection{\sigma}(y)^*\VisibleResponseOperator{\sigma}(y)v,
k
} \\
&=
\ev{
\VisibleResponseOperator{\sigma}(y)v,
\SubstrateProjection{\sigma}(y)k
} \\
&= 0.
\end{align*}
For item (3), if \(u\in\SubstrateCarrier{\sigma}\), set
\[
k
\equiv
u-\CanonicalLift{\sigma}(y)\bigl(\SubstrateProjection{\sigma}(y)u\bigr).
\]
Applying \(\SubstrateProjection{\sigma}(y)\) and using item (1) gives \(\SubstrateProjection{\sigma}(y)k=0\), so \(k\in\SubstrateKernel{\sigma}(y)\). Uniqueness follows because item (1) fixes the non-kernel part. Finally, item (4) is obtained by expanding
\[
u=\CanonicalLift{\sigma}(y)v+k
\]
and using item (2).
\end{proof}

\begin{theorem}[Projection-operator core collapse to a canonical minimal visible-response representative]
\label{thm:main}

    \item the fixed displacement body together with \(\VisibleResponseOperator{\sigma}\) defines a benchmark-faithful canonical minimal visible-response representative \(\MinimalReducedStructure_{\sigma}\) in the sense of Definition \ref{def:minimal};
    \item every point of the full substrate carrier, and in particular every point of the recorded section-centered substrate body, decomposes uniquely into the canonical visible representative of the same displacement datum together with a kernel direction, and the excess contribution beyond the visible representative is exactly the nonnegative kernel-energy term of Proposition \ref{prop:decomp}(4);
    \item the full substrate-body presentation, all retained carrier/kernel directions, and every same-visible-response reduced carrier presentation are benchmark neutral at current scope because the downstream benchmark-facing package already factors through \(\MinimalReducedStructure_{\sigma}\) by Proposition \ref{prop:downstream};
    \item the benchmark boundary remains fixed by Proposition \ref{prop:benchmark};
    \item consequently the next honest burden below this frontier is no longer another replay of the projection-operator core, another restored reduced carrier presentation, or another inert-kernel extension relay. It is to derive or justify the canonical minimal visible-response representative \(\MinimalReducedStructure_{\sigma}\), equivalently the visible-response operator \(\VisibleResponseOperator{\sigma}\) on the fixed section-centered displacement fibers, from still more primitive explicit substrate structure, or to prove a still sharper theorem strictly inside that visible representative.
\end{enumerate}
