\section{Section-Centered Displacement-Body Quadratic Forcing Law}

\begin{definition}[Benchmark-faithful section-centered displacement-body quadratic forcing law]
\label{def:law}
Fix \(\sigma\in\{\Car,\Cur,\Hom\}\). A \emph{benchmark-faithful section-centered displacement-body quadratic forcing law} for sector \(\sigma\) consists of the following data.
\begin{enumerate}
    \item The fixed proto-section benchmark base and fixed section-centered displacement body of Definition \ref{def:route}(1)--(2).
    \item A smooth forcing law
    \[
    \VisibleBodyForcingLaw{\sigma}:
    \PrePreProtoSectionResidualBody{\sigma}\to[0,\infty).
    \]
    \item For every \(y\in\ProtoSectionBody{\sigma}\), the fiber map
    \[
    v\mapsto \VisibleBodyForcingLaw{\sigma}(y,v)
    \]
    is quadratic on \(\PrePreProtoSectionResidualBody{\sigma}(y)\) and satisfies
    \[
    \VisibleBodyForcingLaw{\sigma}(y,v)=0
    \qquad\Longleftrightarrow\qquad
    v=0
    \]
    on that fiber.
    \item Benchmark faithfulness in the sense of Definition \ref{def:route}(4).
\end{enumerate}
\end{definition}

\begin{remark}
Definition \ref{def:law} is intentionally smaller than the current visible-response representative. It records only the quadratic forcing law on the actually realized displacement body. It does \emph{not} yet postulate a positive operator family on the whole displacement vector space \(\PrePreProtoSectionVertical{\sigma}\). The theorem below recovers that operator canonically from this smaller body-level law.
\end{remark}

\begin{proposition}[A body-level quadratic forcing law determines a unique visible-response operator]
\label{prop:operator}
Fix a benchmark-faithful section-centered displacement-body quadratic forcing law for sector \(\sigma\). Then for every \(y\in\ProtoSectionBody{\sigma}\) there exists a unique positive self-adjoint operator
\[
\VisibleResponseOperator{\sigma}(y):
\PrePreProtoSectionVertical{\sigma}\to\PrePreProtoSectionVertical{\sigma}
\]
such that
\begin{equation}
\VisibleBodyForcingLaw{\sigma}(y,v)
=
\frac{1}{2}
\ev{v,\VisibleResponseOperator{\sigma}(y)v}
\label{eq:operator}
\end{equation}
for every \(v\in\PrePreProtoSectionResidualBody{\sigma}(y)\). Moreover the family \(y\mapsto \VisibleResponseOperator{\sigma}(y)\) is smooth.
\end{proposition}

\begin{proof}
Fix \(y\in\ProtoSectionBody{\sigma}\). By Definition \ref{def:route}(2), the fiber \(\PrePreProtoSectionResidualBody{\sigma}(y)\) is compact, convex, contains \(0\), and has nonempty interior in the finite-dimensional vector space \(\PrePreProtoSectionVertical{\sigma}\). By Definition \ref{def:law}(3), the map
\[
v\mapsto \VisibleBodyForcingLaw{\sigma}(y,v)
\]
is quadratic on that set. A quadratic function on a convex set with nonempty interior extends uniquely to a quadratic form on the whole vector space. By polarization and the Euclidean inner product on \(\PrePreProtoSectionVertical{\sigma}\), there is therefore a unique self-adjoint operator \(\VisibleResponseOperator{\sigma}(y)\) satisfying \eqref{eq:operator}.

Positivity follows because \(\VisibleBodyForcingLaw{\sigma}(y,v)\ge 0\) and vanishes only at \(v=0\) on a set with nonempty interior, so the extended quadratic form is positive definite. Smoothness in \(y\) follows from the smooth dependence of \(\VisibleBodyForcingLaw{\sigma}\) on \(y\) and the uniqueness of the quadratic-form coefficients.
\end{proof}

\begin{theorem}[Canonical minimal visible-response representative from a smaller body-level law]
\label{thm:main}

\begin{enumerate}
    \item the body-level law determines the unique positive visible-response operator \(\VisibleResponseOperator{\sigma}\) on the full displacement fibers by Proposition \ref{prop:operator};
    \item the fixed displacement body together with \(\VisibleResponseOperator{\sigma}\) defines the canonical minimal visible-response representative \(\MinimalReducedStructure_{\sigma}\) of Definition \ref{def:minimal};
    \item the induced current quadratic response on the fixed displacement body is exactly the original law,

    \frac{1}{2}\ev{v,\VisibleResponseOperator{\sigma}(y)v}

    \VisibleBodyForcingLaw{\sigma}(y,v);

    \item the benchmark one-metric package \eqref{eq:fixedoutputs} is preserved, the ordinary matter route is unchanged, there is no second operative metric, and the low-energy matter law is not enlarged;

\end{theorem}

\begin{proof}

For item (4), Proposition \ref{prop:frontier}(2) says that all current downstream benchmark-facing uses already factor through the visible-response representative on the fixed displacement body.
