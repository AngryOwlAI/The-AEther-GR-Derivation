\section{Canonical Recovery of the Current Section Core}

\begin{proposition}[The canonical proto-section minimizer is well defined]
\label{prop:minimizer}
For every benchmark-faithful proto-section dynamics package, the map \(\ProtoSectionMin{\sigma}\) of Definition \ref{def:induced} is a smooth section of the composite projection \(\ProtoProj{\sigma}\circ\ProtoSectionProj{\sigma}\), so that
\[
\ProtoProj{\sigma}\circ\ProtoSectionProj{\sigma}\circ\ProtoSectionMin{\sigma}
=
\mathrm{id}_{\GermBody{\sigma}}.
\]
\end{proposition}

\begin{proof}
Definition \ref{def:protosection}(1) makes each fiber \(\ProtoSectionFiber{\sigma}{x}\) nonempty, compact, and convex. Definition \ref{def:protosection}(2) states that \(\ProtoSectionFunctional{\sigma}\) has a unique minimizer on each such fiber and that the minimizer depends smoothly on \(x\). Definition \ref{def:induced} therefore gives a well-defined smooth map
\[
\ProtoSectionMin{\sigma}:\GermBody{\sigma}\to\ProtoSectionBody{\sigma}.
\]
Because every minimizer lies in the fiber over the point at which it is selected, we have
\[
\ProtoProj{\sigma}\Bigl(
\ProtoSectionProj{\sigma}\bigl(\ProtoSectionMin{\sigma}(x)\bigr)
\Bigr)=x
\]
for every \(x\in\GermBody{\sigma}\), which is the displayed identity.
\end{proof}

\begin{proposition}[The exact proto-section shell is the shell restriction of the canonical minimizer]
\label{prop:shell}
For every benchmark-faithful proto-section dynamics package,
\[
\ProtoSectionShell{\sigma}
=
\ProtoSectionMin{\sigma}\bigl(\GermShell{\sigma}\bigr),
\]
and the shell restriction
\[
\ProtoSectionMin{\sigma}\big|_{\GermShell{\sigma}}
:
\GermShell{\sigma}\xrightarrow{\cong}\ProtoSectionShell{\sigma}
\]
is a diffeomorphism with inverse
\[
\ProtoProj{\sigma}\circ\ProtoSectionProj{\sigma}\big|_{\ProtoSectionShell{\sigma}}
:
\ProtoSectionShell{\sigma}\xrightarrow{\cong}\GermShell{\sigma}.
\]
\end{proposition}

\begin{proof}
Let \(x\in\GermShell{\sigma}\). By Definition \ref{def:induced}, \(\ProtoSectionMin{\sigma}(x)\) is the unique minimizer of \(\ProtoSectionFunctional{\sigma}\) on the fiber \(\ProtoSectionFiber{\sigma}{x}\). Since \(x\in\GermShell{\sigma}\), Definition \ref{def:protosection}(3) implies
\[
\ProtoSectionMin{\sigma}(x)\in\ProtoSectionShell{\sigma}.
\]
Hence
\[
\ProtoSectionMin{\sigma}\bigl(\GermShell{\sigma}\bigr)\subset\ProtoSectionShell{\sigma}.
\]

Conversely, if \(y\in\ProtoSectionShell{\sigma}\), then
\[
x\equiv\ProtoProj{\sigma}\bigl(\ProtoSectionProj{\sigma}(y)\bigr)
\]
lies in \(\GermShell{\sigma}\), and Definition \ref{def:protosection}(3) says that \(y\) is a minimizer of \(\ProtoSectionFunctional{\sigma}\) on \(\ProtoSectionFiber{\sigma}{x}\). By uniqueness of that minimizer, \(y=\ProtoSectionMin{\sigma}(x)\). This proves the displayed equality.

Definition \ref{def:protosection}(3) gives a diffeomorphism
\[
\ProtoSectionProj{\sigma}\big|_{\ProtoSectionShell{\sigma}}
:
\ProtoSectionShell{\sigma}\xrightarrow{\cong}\ProtoShell{\sigma}.
\]
Definition \ref{def:core}(2) gives a diffeomorphism
\[
\ProtoProj{\sigma}\big|_{\ProtoShell{\sigma}}
:
\ProtoShell{\sigma}\xrightarrow{\cong}\GermShell{\sigma}.
\]
Their composition is therefore a diffeomorphism
\[
\ProtoProj{\sigma}\circ\ProtoSectionProj{\sigma}\big|_{\ProtoSectionShell{\sigma}}
:
\ProtoSectionShell{\sigma}\xrightarrow{\cong}\GermShell{\sigma},
\]
and the displayed equality shows that its inverse is exactly \(\ProtoSectionMin{\sigma}\big|_{\GermShell{\sigma}}\).
\end{proof}

\begin{proposition}[The projected shell transport reproduces the current section core]
\label{prop:sectioncore}
For every benchmark-faithful proto-section dynamics package,
\[
\InducedTransportSec{\sigma}
=
\left(
\ProtoProj{\sigma}\big|_{\ProtoShell{\sigma}}
\right)^{-1}
=
\ProtoTransportSec{\sigma}
\qquad
\text{on }\GermShell{\sigma}.
\]
\end{proposition}

\begin{proof}
Fix \(x\in\GermShell{\sigma}\). By Definition \ref{def:induced},
\[
\InducedTransportSec{\sigma}(x)
=
\ProtoSectionProj{\sigma}\bigl(\ProtoSectionMin{\sigma}(x)\bigr).
\]
By Proposition \ref{prop:shell}, \(\ProtoSectionMin{\sigma}(x)\in\ProtoSectionShell{\sigma}\), so the displayed point lies in \(\ProtoShell{\sigma}\). Applying \(\ProtoProj{\sigma}\) gives
\[
\ProtoProj{\sigma}\bigl(\InducedTransportSec{\sigma}(x)\bigr)
=
\ProtoProj{\sigma}\Bigl(
\ProtoSectionProj{\sigma}\bigl(\ProtoSectionMin{\sigma}(x)\bigr)
\Bigr)
=
x
\]
by Proposition \ref{prop:minimizer}. Therefore \(\InducedTransportSec{\sigma}(x)\) is a point of \(\ProtoShell{\sigma}\) projecting to \(x\in\GermShell{\sigma}\). Since
\[
\ProtoProj{\sigma}\big|_{\ProtoShell{\sigma}}
:
\ProtoShell{\sigma}\xrightarrow{\cong}\GermShell{\sigma}
\]
is a diffeomorphism by Definition \ref{def:core}(2), that point is uniquely determined and equals
\[
\left(
\ProtoProj{\sigma}\big|_{\ProtoShell{\sigma}}
\right)^{-1}(x)
=
\ProtoTransportSec{\sigma}(x).
\]
\end{proof}

\begin{proposition}[The projected shell transport is canonical]
\label{prop:canonical}
Fix a benchmark-faithful proto-section dynamics package. If
\[
s:\GermShell{\sigma}\to\ProtoSectionBody{\sigma}
\]
is any smooth map satisfying
\[
s(x)\in\ProtoSectionFiber{\sigma}{x},
\qquad
\ProtoSectionFunctional{\sigma}\bigl(s(x)\bigr)
=
\min_{y\in\ProtoSectionFiber{\sigma}{x}}
\ProtoSectionFunctional{\sigma}(y)
\]
for every \(x\in\GermShell{\sigma}\), then
\[
s=\ProtoSectionMin{\sigma}\big|_{\GermShell{\sigma}}
\]
and
\[
\ProtoSectionProj{\sigma}\circ s
=
\InducedTransportSec{\sigma}
=
\ProtoTransportSec{\sigma}.
\]
\end{proposition}

\begin{proof}
For each \(x\in\GermShell{\sigma}\), Definition \ref{def:protosection}(2) supplies a unique minimizer of \(\ProtoSectionFunctional{\sigma}\) on \(\ProtoSectionFiber{\sigma}{x}\). Therefore the assumed map \(s\) must agree pointwise with the unique minimizer \(\ProtoSectionMin{\sigma}(x)\). The projected identity then follows from Proposition \ref{prop:sectioncore}.
\end{proof}
