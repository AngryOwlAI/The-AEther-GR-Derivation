\begin{lemma}[The stationary Hessian determines a unique spectral generator]
\label{lem:riesz}
Assume Definition \ref{def:stationary}. Then there exists a unique self-adjoint positive semidefinite operator
\[
\SpecOp{\sigma}:\SpecSpace{\sigma}\to\SpecSpace{\sigma}
\]
satisfying \eqref{eq:riesz}. Moreover,
\[
\ker\SpecOp{\sigma}=\StatZero{\sigma},
\qquad
\operatorname{im}\SpecOp{\sigma}=\StatHidden{\sigma},
\]
and
\[
\ev{\eta,\SpecOp{\sigma}\eta}
=
\StatHess{\sigma}(\eta,\eta)
\ge
\lambda_{\sigma}^{\mathrm{st}}\|\eta\|^{2}
\qquad
\text{for all }\eta\in\StatHidden{\sigma}.
\]
\end{lemma}

\begin{proof}
Because \(\StatHess{\sigma}\) is a symmetric bilinear form on the finite-dimensional Euclidean space \(\SpecSpace{\sigma}=\StatSpace{\sigma}\), the Riesz representation theorem yields a unique self-adjoint operator \(\SpecOp{\sigma}\) satisfying \eqref{eq:riesz}. Positivity follows from \eqref{eq:hesszero} and \eqref{eq:stgap}: if \(u=z+\eta\) with \(z\in\StatZero{\sigma}\) and \(\eta\in\StatHidden{\sigma}\), then
\[
\ev{u,\SpecOp{\sigma}u}
=
\StatHess{\sigma}(u,u)
=
\StatHess{\sigma}(\eta,\eta)
\ge
0.
\]
If \(z\in\StatZero{\sigma}\), then \eqref{eq:hesszero} gives
\[
\ev{u,\SpecOp{\sigma}z}
=
\StatHess{\sigma}(u,z)=0
\qquad

\]

=
\bigl(\ker\SpecOp{\sigma}\bigr)^{\perp}

\StatHidden{\sigma}.
\]
The hidden-sector lower bound is exactly \eqref{eq:stgap}.
\end{proof}
