\section{Canonical Recovery of the Current Selector Package}

\begin{proposition}[The canonical pre-selector minimizer section is well defined]
\label{prop:section}
For every benchmark-faithful pre-selector dynamics package, the map \(\PreSelectorSelectorMin{\sigma}\) of Definition \ref{def:presel} is a smooth section of \(\PreSelectorProj{\sigma}\), so that
\[
\PreSelectorProj{\sigma}\circ\PreSelectorSelectorMin{\sigma}
=
\mathrm{id}_{\YoctoSelectorBody{\sigma}}.
\]
\end{proposition}

\begin{proof}
Definition \ref{def:presel}(1) makes each fiber \(\PreSelectorFiber{\sigma}{y}\) nonempty, compact, and convex. Definition \ref{def:presel}(2) states that \(\PreSelectorFunctional{\sigma}\) has a unique minimizer on each such fiber and that the minimizer depends smoothly on \(y\). Definition \ref{def:presel}(2) therefore gives a well-defined smooth map
\[
\PreSelectorSelectorMin{\sigma}:\YoctoSelectorBody{\sigma}\to\PreSelectorBody{\sigma}.
\]
Because every minimizer lies in the fiber over the point at which it is selected, we have
\[
\PreSelectorProj{\sigma}\bigl(\PreSelectorSelectorMin{\sigma}(y)\bigr)=y
\]
for every \(y\in\YoctoSelectorBody{\sigma}\), which is the displayed identity.
\end{proof}

\begin{proposition}[The exact selector shell is the projected exact pre-selector shell]
\label{prop:shell}
For every benchmark-faithful pre-selector dynamics package,
\[
\PreSelectorProj{\sigma}\bigl(\PreSelectorShell{\sigma}\bigr)
=
\YoctoSelectorShell{\sigma},
\]
and the shell restriction
\[
\PreSelectorProj{\sigma}\big|_{\PreSelectorShell{\sigma}}
:
\PreSelectorShell{\sigma}\xrightarrow{\cong}\YoctoSelectorShell{\sigma}
\]
is a diffeomorphism with inverse
\[
\PreSelectorSelectorMin{\sigma}\big|_{\YoctoSelectorShell{\sigma}}
:
\YoctoSelectorShell{\sigma}\xrightarrow{\cong}\PreSelectorShell{\sigma}.
\]
\end{proposition}

\begin{proof}
Definition \ref{def:presel}(3) states directly that
\[
\PreSelectorProj{\sigma}\big|_{\PreSelectorShell{\sigma}}
:
\PreSelectorShell{\sigma}\xrightarrow{\cong}\YoctoSelectorShell{\sigma}
\]
is a diffeomorphism, so the image identity follows immediately. To identify the inverse, let \(y\in\YoctoSelectorShell{\sigma}\). By Proposition \ref{prop:section}, \(\PreSelectorSelectorMin{\sigma}(y)\) lies in the fiber over \(y\). Since \(y\in\YoctoSelectorShell{\sigma}\), Definition \ref{def:presel}(3) shows that this point belongs to \(\PreSelectorShell{\sigma}\). Hence the shell restriction of \(\PreSelectorSelectorMin{\sigma}\) is indeed the inverse diffeomorphism.
\end{proof}
