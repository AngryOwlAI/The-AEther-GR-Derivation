\begin{definition}[Residual-law-induced hidden-response package]
\label{def:induced}

\end{equation}
for every \((y,v)\in\PrePreProtoSectionResidualBody{\sigma}\).
\end{definition}

\begin{proposition}[The pullback residual operator induces a positive hidden Hessian]
\label{prop:pullback}
Under Definition \ref{def:induced}, the family \(\HiddenHessian{\sigma}(y)\) is positive self-adjoint and positive definite on \(\HiddenCarrier{\sigma}\) for every \(y\in\ProtoSectionBody{\sigma}\).
\end{proposition}

\begin{proof}
Self-adjointness is immediate from the definition
\[
\HiddenHessian{\sigma}(y)
=
\ResidualOperator{\sigma}(y)^*\ResidualOperator{\sigma}(y).
\]
For any nonzero \(u\in\HiddenCarrier{\sigma}=\ResidualState{\sigma}\),
\[
\ev{u,\HiddenHessian{\sigma}(y)u}
=
\ev{
\ResidualOperator{\sigma}(y)u,
\ResidualOperator{\sigma}(y)u
}
=
\norm{\ResidualOperator{\sigma}(y)u}^{2}.
\]
Because \(\ResidualOperator{\sigma}(y)\) is injective, the last quantity is strictly positive whenever \(u\neq 0\). Hence \(\HiddenHessian{\sigma}(y)\) is positive definite.
\end{proof}

\label{prop:leastsq}

\ \HiddenReadout{\sigma}(y)u=v

\begin{proof}

\[

\ev{

\ResidualOperator{\sigma}(y)^*\ResidualOperator{\sigma}(y)u

\frac{1}{2}\norm{\ResidualOperator{\sigma}(y)u}^{2}.
\]
Because the readout families agree by Definition \ref{def:induced}, the admissible sets in \eqref{eq:residualenergy} and \eqref{eq:leastsq} are the same. Therefore the two minimization problems coincide exactly.
