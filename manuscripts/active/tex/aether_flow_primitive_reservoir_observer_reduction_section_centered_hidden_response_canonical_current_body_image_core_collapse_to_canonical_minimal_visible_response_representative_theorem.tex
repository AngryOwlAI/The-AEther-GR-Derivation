\section{Visible Response Readout}

\begin{proposition}[The full current image core determines a unique visible-response operator]
\label{prop:visible}
For each \(y\in\ProtoSectionBody{\sigma}\), there exists a unique positive self-adjoint operator
\[
\VisibleResponseOperator{\sigma}(y):
\PrePreProtoSectionVertical{\sigma}\to\PrePreProtoSectionVertical{\sigma}
\]
such that
\begin{equation}
\CoreForcedBodyResponse{\sigma}(y,v)
=
\frac{1}{2}
\ev{v,\VisibleResponseOperator{\sigma}(y)v}
\label{eq:visible}
\end{equation}
for every \((y,v)\in\PrePreProtoSectionResidualBody{\sigma}\). The family \(y\mapsto\VisibleResponseOperator{\sigma}(y)\) is smooth.
\end{proposition}

\begin{proof}
Fix \(y\in\ProtoSectionBody{\sigma}\). By Definition \ref{def:frontier}(3), the function
\[
v\mapsto \CoreForcedBodyResponse{\sigma}(y,v)
\]
is quadratic on the compact convex body \(\PrePreProtoSectionResidualBody{\sigma}(y)\), which contains \(0\) and has nonempty interior in the finite-dimensional vector space \(\PrePreProtoSectionVertical{\sigma}\). A quadratic function on such a set extends uniquely to a quadratic form on the whole vector space. By polarization and the Euclidean inner product on \(\PrePreProtoSectionVertical{\sigma}\), there is therefore a unique self-adjoint operator \(\VisibleResponseOperator{\sigma}(y)\) satisfying \eqref{eq:visible}.

Positivity follows because the fixed body-response route is nonnegative and vanishes only at \(v=0\). Hence the extended quadratic form is positive definite. Smoothness in \(y\) follows from the smoothness of \(\CoreForcedBodyResponse{\sigma}\) in Definition \ref{def:frontier}(3).
\end{proof}

\begin{remark}
Proposition \ref{prop:visible} is the precise compression below the full current-image-core frontier. The full image-core package determines the visible quadratic response on the fixed displacement fibers, and at current benchmark-facing scope that visible quadratic response is the only datum that remains directly operative there.
\end{remark}

\begin{proposition}[The image-core response law is exactly the pullback of the visible quadratic response]
\label{prop:factor}
For every \(y\in\ProtoSectionBody{\sigma}\) and every \(u\in\CanonicalImageBody{\sigma}(y)\),
\begin{equation}
\CanonicalImageResponseLaw{\sigma}(y,u)
=
\frac{1}{2}
\ev{
\ImageProjection{\sigma}(y)u,
\VisibleResponseOperator{\sigma}(y)\ImageProjection{\sigma}(y)u
}.
\label{eq:imagefactor}
\end{equation}
Moreover,
\begin{equation}
\CanonicalImageShell{\sigma}
=
\left\{
(y,u)\in\ProtoSectionBody{\sigma}\times\CanonicalImageBody{\sigma}(y):
\ImageProjection{\sigma}(y)u=0
\right\}.
\label{eq:shellfactor}
\end{equation}
\end{proposition}

\begin{proof}
Fix \(y\in\ProtoSectionBody{\sigma}\) and \(u\in\CanonicalImageBody{\sigma}(y)\). Set
\[
v\equiv \ImageProjection{\sigma}(y)u
\in
\PrePreProtoSectionResidualBody{\sigma}(y).
\]
Because \(\ImageProjection{\sigma}(y)\) is inverse to \(\CanonicalLift{\sigma}(y)\) on the fixed image body, we have \(u=\CanonicalLift{\sigma}(y)v\). Therefore \eqref{eq:imagepullback} and Proposition \ref{prop:visible} give
\[
\CanonicalImageResponseLaw{\sigma}(y,u)
=
\CoreForcedBodyResponse{\sigma}(y,v)
=
\frac{1}{2}
\ev{v,\VisibleResponseOperator{\sigma}(y)v},
\]
which is exactly \eqref{eq:imagefactor}.

For \eqref{eq:shellfactor}, first suppose \((y,u)\in\CanonicalImageShell{\sigma}\). By Definition \ref{def:frontier}(4), we have \(y\in\ProtoSectionShell{\sigma}\) and \(u=\CanonicalLift{\sigma}(y)0\). Hence \(\ImageProjection{\sigma}(y)u=0\).

Conversely, suppose \(u\in\CanonicalImageBody{\sigma}(y)\) and \(\ImageProjection{\sigma}(y)u=0\). Because \(\ImageProjection{\sigma}(y)\) is inverse to \(\CanonicalLift{\sigma}(y)\) on the fixed image body, we get \(u=\CanonicalLift{\sigma}(y)0\). Since \(0\in\PrePreProtoSectionResidualBody{\sigma}(y)\) exactly on \(\ProtoSectionShell{\sigma}\) by Definition \ref{def:frontier}(2), it follows that \(y\in\ProtoSectionShell{\sigma}\). Therefore \((y,u)\in\CanonicalImageShell{\sigma}\).
\end{proof}
