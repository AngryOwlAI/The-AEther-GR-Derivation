\section{Recovery of the Current Pre-Selector Selector Dynamics}

\begin{proposition}[The inner kernel minimizer lift is well defined]
\label{prop:inner}
For every benchmark-faithful kernel package, the minimizer lift \(\KernelMinLift{\sigma}\) of Definition \ref{def:kernel} is a smooth section of \(\KernelProj{\sigma}\), so that
\[
\KernelProj{\sigma}\circ\KernelMinLift{\sigma}
=
\mathrm{id}_{\PreSectionSelectorBody{\sigma}}.
\]
In particular, the induced outer selector functional \(\InducedSelectorFunctional{\sigma}\) is a well-defined smooth function on \(\PreSectionSelectorSpace{\sigma}\) whose values on \(\PreSectionSelectorBody{\sigma}\) are the effective minimum values of the deeper kernel functional on the inner kernel fibers.
\end{proposition}

\begin{proof}
Definition \ref{def:kernel}(1) makes every inner kernel fiber \(\KernelFiber{\sigma}{u}\) nonempty, compact, and convex. Definition \ref{def:kernel}(2) states that \(\KernelFunctional{\sigma}\) has a unique minimizer on each such fiber and that the minimizer depends smoothly on \(u\). Therefore \(\KernelMinLift{\sigma}\) is a well-defined smooth map on \(\PreSectionSelectorBody{\sigma}\). Each chosen minimizer lies in the fiber over the point at which it is selected, so
\[
\KernelProj{\sigma}\bigl(\KernelMinLift{\sigma}(u)\bigr)=u
\]
for every \(u\in\PreSectionSelectorBody{\sigma}\). The remaining smoothness statement for \(\InducedSelectorFunctional{\sigma}\) on \(\PreSectionSelectorSpace{\sigma}\) is part of Definition \ref{def:kernel}(2), and the displayed formula identifies its values on \(\PreSectionSelectorBody{\sigma}\) with the effective minimum values.
\end{proof}

\begin{proposition}[Composed kernel minimization factors through the induced selector minimization]
\label{prop:factor}
For every benchmark-faithful kernel package, define the induced outer minimizer section
\[
\InducedPreSelectorMinSection{\sigma}(y)
\equiv
\operatorname*{arg\,min}_{u\in\PreSectionSelectorFiber{\sigma}{y}}
\InducedSelectorFunctional{\sigma}(u),
\qquad
y\in\YoctoSelectorBody{\sigma}.
\]
Then
\[
\KernelComposedMin{\sigma}(y)
=
\KernelMinLift{\sigma}\bigl(\InducedPreSelectorMinSection{\sigma}(y)\bigr)
\]
for every \(y\in\YoctoSelectorBody{\sigma}\), and hence
\[
\KernelProj{\sigma}\bigl(\KernelComposedMin{\sigma}(y)\bigr)
=
\InducedPreSelectorMinSection{\sigma}(y).
\]
\end{proposition}

\begin{proof}
Fix \(y\in\YoctoSelectorBody{\sigma}\). Every \(w\in\KernelComposedFiber{\sigma}{y}\) satisfies \(u\equiv\KernelProj{\sigma}(w)\in\PreSectionSelectorFiber{\sigma}{y}\). By Proposition \ref{prop:inner},
\[
\KernelFunctional{\sigma}(w)
\geq
\KernelFunctional{\sigma}\bigl(\KernelMinLift{\sigma}(u)\bigr)
=
\InducedSelectorFunctional{\sigma}(u).
\]
By Definition \ref{def:kernel}(2), the induced outer functional is strictly convex on \(\PreSectionSelectorFiber{\sigma}{y}\) and therefore has a unique minimizer \(\InducedPreSelectorMinSection{\sigma}(y)\). Consequently, for every \(w\in\KernelComposedFiber{\sigma}{y}\),
\[
\KernelFunctional{\sigma}(w)
\geq
\InducedSelectorFunctional{\sigma}\bigl(\InducedPreSelectorMinSection{\sigma}(y)\bigr)
=
\KernelFunctional{\sigma}\bigl(\KernelMinLift{\sigma}(\InducedPreSelectorMinSection{\sigma}(y))\bigr).
\]
So \(\KernelMinLift{\sigma}(\InducedPreSelectorMinSection{\sigma}(y))\) is a minimizer of \(\KernelFunctional{\sigma}\) on the composed kernel fiber \(\KernelComposedFiber{\sigma}{y}\). Definition \ref{def:kernel}(3) says that minimizer is unique, hence the first displayed identity follows. The second identity is then Proposition \ref{prop:inner}.
\end{proof}

\begin{proposition}[The exact selector shell is the projected exact kernel shell]
\label{prop:shell}

\[
\InducedSelectorShell{\sigma}

\]
and the restriction
\[
\KernelProj{\sigma}\big|_{\KernelShell{\sigma}}
:
\KernelShell{\sigma}\xrightarrow{\cong}\InducedSelectorShell{\sigma}
\]
is a diffeomorphism. Moreover,
\[
\PreSectionSelectorProj{\sigma}\big|_{\InducedSelectorShell{\sigma}}

is a diffeomorphism.
\end{proposition}

\begin{proof}
The first diffeomorphism is part of Definition \ref{def:kernel}(3), so it remains only to prove the final statement. Let \(u\in\InducedSelectorShell{\sigma}\). Then \(\PreSectionSelectorProj{\sigma}(u)\in\YoctoSelectorShell{\sigma}\) by definition. Conversely, if \(y\in\YoctoSelectorShell{\sigma}\), then Proposition \ref{prop:factor} gives a unique composed-fiber minimizer \(\KernelComposedMin{\sigma}(y)\in\KernelShell{\sigma}\), hence
\[
u\equiv\KernelProj{\sigma}\bigl(\KernelComposedMin{\sigma}(y)\bigr)\in\InducedSelectorShell{\sigma}
\]
and \(\PreSectionSelectorProj{\sigma}(u)=y\). This proves surjectivity. If \(u_1,u_2\in\InducedSelectorShell{\sigma}\) have the same image \(y\in\YoctoSelectorShell{\sigma}\), then both are outer projections of the unique composed-fiber minimizer \(\KernelComposedMin{\sigma}(y)\) by Proposition \ref{prop:factor}, so \(u_1=u_2\). Smoothness of the inverse follows from smooth dependence of \(\KernelComposedMin{\sigma}\) on \(y\) and smoothness of \(\KernelProj{\sigma}\).
\end{proof}
