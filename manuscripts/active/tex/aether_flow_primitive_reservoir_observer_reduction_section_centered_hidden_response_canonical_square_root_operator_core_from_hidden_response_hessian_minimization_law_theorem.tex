\section{Canonical Reduced Readout}

\begin{proposition}[Positive reduced Schur operator]
\label{prop:schur}
Fix a benchmark-faithful hidden-response Hessian minimization law for sector \(\sigma\). For each \(y\in\ProtoSectionBody{\sigma}\), define
\begin{equation}
\ReducedSchur{\sigma}(y)
\equiv
\HiddenReadout{\sigma}(y)
\HiddenHessian{\sigma}(y)^{-1}
\HiddenReadout{\sigma}(y)^*
:
\PrePreProtoSectionVertical{\sigma}\to\PrePreProtoSectionVertical{\sigma},
\label{eq:schur}
\end{equation}
where \((\cdot)^*\) denotes Euclidean adjoint with respect to the fixed inner products on \(\HiddenCarrier{\sigma}\) and \(\PrePreProtoSectionVertical{\sigma}\). Then \(\ReducedSchur{\sigma}(y)\) is positive self-adjoint and invertible.
\end{proposition}

\begin{proof}
Self-adjointness is immediate from the self-adjointness of \(\HiddenHessian{\sigma}(y)^{-1}\). For any \(0\neq w\in\PrePreProtoSectionVertical{\sigma}\),
\[
\ev{w,\ReducedSchur{\sigma}(y)w}
=
\ev{
\HiddenHessian{\sigma}(y)^{-1/2}\HiddenReadout{\sigma}(y)^*w,
\HiddenHessian{\sigma}(y)^{-1/2}\HiddenReadout{\sigma}(y)^*w
}
>0
\]
because \(\HiddenReadout{\sigma}(y)\) is surjective and therefore \(\HiddenReadout{\sigma}(y)^*\) is injective. Hence \(\ReducedSchur{\sigma}(y)\) is positive definite and therefore invertible on the finite-dimensional space \(\PrePreProtoSectionVertical{\sigma}\).
\end{proof}

\begin{proposition}[Canonical minimizer lift properties]
\label{prop:canonical}
For every \(y\in\ProtoSectionBody{\sigma}\),
\begin{enumerate}
    \item
    \[
    \HiddenReadout{\sigma}(y)\circ\HiddenMinLift{\sigma}(y)
    =
    \mathrm{id}_{\PrePreProtoSectionVertical{\sigma}};
    \]
    \item if \(\HiddenKernel{\sigma}(y)\equiv\ker\HiddenReadout{\sigma}(y)\), then
    \[
    \ev{
    \HiddenMinLift{\sigma}(y)v,
    \HiddenHessian{\sigma}(y)k
    }
    =
    0
    \]
    for every \(v\in\PrePreProtoSectionVertical{\sigma}\) and every \(k\in\HiddenKernel{\sigma}(y)\);
    \item every \(u\in\HiddenCarrier{\sigma}\) admits a unique decomposition
    \[
    u
    =
    \HiddenMinLift{\sigma}(y)\bigl(\HiddenReadout{\sigma}(y)u\bigr)
    +
    k,
    \qquad
    k\in\HiddenKernel{\sigma}(y);
    \]
    \item \(\HiddenMinLift{\sigma}(y)\) is linear in the displacement argument and smooth in \(y\).
\end{enumerate}
\end{proposition}

\begin{proof}
Item (1) is
\[
\HiddenReadout{\sigma}(y)\HiddenMinLift{\sigma}(y)
=
\HiddenReadout{\sigma}(y)
\HiddenHessian{\sigma}(y)^{-1}
\HiddenReadout{\sigma}(y)^*
\ReducedSchur{\sigma}(y)^{-1}
=
\ReducedSchur{\sigma}(y)\ReducedSchur{\sigma}(y)^{-1}
=
\mathrm{id}.
\]
For item (2), let \(k\in\HiddenKernel{\sigma}(y)\). Then
\[
\ev{
\HiddenMinLift{\sigma}(y)v,
\HiddenHessian{\sigma}(y)k
}
=
\ev{
\ReducedSchur{\sigma}(y)^{-1}v,
\HiddenReadout{\sigma}(y)k
}
=
0.
\]
Item (3) follows by setting
\[
k
\equiv
u-\HiddenMinLift{\sigma}(y)\bigl(\HiddenReadout{\sigma}(y)u\bigr)
\]
and applying item (1). Uniqueness follows because item (1) fixes the non-kernel part. Linearity and smoothness are immediate from \eqref{eq:lift}.
\end{proof}

\begin{proposition}[Unique hidden-response energy minimizer]
\label{prop:min}
For every \(y\in\ProtoSectionBody{\sigma}\) and every \(v\in\PrePreProtoSectionVertical{\sigma}\), the canonical hidden-response minimizer lift \(\HiddenMinLift{\sigma}(y)v\) is the unique minimizer of
\begin{equation}
\min\left\{
\frac{1}{2}\ev{u,\HiddenHessian{\sigma}(y)u}:
u\in\HiddenCarrier{\sigma},
\ \HiddenReadout{\sigma}(y)u=v
\right\}.
\label{eq:minproblem}
\end{equation}
\end{proposition}

\begin{proof}
Let \(u\in\HiddenCarrier{\sigma}\) satisfy \(\HiddenReadout{\sigma}(y)u=v\). By Proposition \ref{prop:canonical}(3),
\[
u=\HiddenMinLift{\sigma}(y)v+k,
\qquad
k\in\HiddenKernel{\sigma}(y).
\]
Using Proposition \ref{prop:canonical}(2),
\[
\ev{u,\HiddenHessian{\sigma}(y)u}
=
\ev{
\HiddenMinLift{\sigma}(y)v,
\HiddenHessian{\sigma}(y)\HiddenMinLift{\sigma}(y)v
}
+
\ev{k,\HiddenHessian{\sigma}(y)k}.
\]
The last term is strictly positive for every nonzero \(k\) because \(\HiddenHessian{\sigma}(y)\) is positive definite. Therefore the unique minimizer of \eqref{eq:minproblem} is \(\HiddenMinLift{\sigma}(y)v\).
\end{proof}
